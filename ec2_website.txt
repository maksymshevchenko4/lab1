pipeline {
agent any
        stages {
            stage('checkout') {
                steps {
                    git branch: 'main', url: 'https://github.com/maksymshevchenko4/lab1.git'
                }
            }
            stage('terraform action') {
                steps {
                    echo "Terraform action is --> ${action}"
                    sh '''
                    cd ec2_website
                    ls
                    terraform init
                    terraform plan
                    terraform ${action} -auto-approve '''
                }
            }
        }
}
